 30: /**
 31:  * A web oriented widget is intended to dynamically build instances of
 32:  * HTML elements that can than be rendered throughout the view.
 33:  *
 34:  * @author Lumina Framework <lumina@incubator.neticle.com>
 35:  * @since 0.2.0
 36:  */
 37: abstract class Widget extends Element
 38: {
 39:     /**
 40:      * The next auto incrementable widget id.
 41:      *
 42:      * @type int
 43:      */
 44:     private static $nextWidgetId = 1;
 45: 
 46:     /**
 47:      * The unique widget string identifier.
 48:      *
 49:      * @type string
 50:      */
 51:     private $id;
 52: 
 53:     /**
 54:      * An associative array of currently registered widget classes, indexed
 55:      * by name.
 56:      *
 57:      * @type array
 58:      */
 59:     private static $widgetClasses =
 60:     [
 61:         'document' => 'system\\web\\widget\\DocumentWidget',
 62:         'data.grid' => 'system\\web\\widget\\data\\grid\\GridWidget',
 63:         'data.paginator' => 'system\\web\\widget\\data\\PaginatorWidget',
 64:         'navigation.breadcrumb' => 'system\\web\\widget\\navigation\\BreadcrumbWidget',
 65:         'navigation.button' => 'system\\web\\widget\\navigation\\ButtonWidget',
 66:         'navigation.dropDownButton' => 'system\\web\\widget\\navigation\\DropDownButtonWidget'
 67:     ];
 68:     
 69:     /**
 70:      * Creates a new widget instance.
 71:      *
 72:      * This method provides a convenient way to quickly replace the behavior
 73:      * of any widget by defining a new class for its name.
 74:      *
 75:      * @param string $name
 76:      *  The name of the widget to create the instance of.
 77:      *
 78:      * @param mixed $...
 79:      *  The widget constructor arguments.
 80:      */
 81:     public static function create($name)
 82:     {
 83:         if (!isset(self::$widgetClasses[$name]))
 84:         {
 85:             throw new RuntimeException('Widget "' . $name . '" is not defined.');
 86:         }
 87:         
 88:         $class = self::$widgetClasses[$name];
 89:         $reflection = new \ReflectionClass($class);
 90:         
 91:         if (!$reflection->isSubclassOf(__CLASS__))
 92:         {
 93:             throw new RuntimeException('Widget "' . $name . '" (class "' . $class . '") does not map to a valid class.');
 94:         }
 95:         
 96:         return $reflection->newInstanceArgs(array_slice(func_get_args(), 1));
 97:     }
 98:     
 99:     /**
100:      * Defines the class for a specific widget, by name.
101:      *
102:      * @param string $name
103:      *  The name of the widget to define the class for.
104:      *
105:      * @param string $class
106:      *  The absolute name of the class to define the widget with.
107:      */
108:     public static function setWidgetClass($name, $class)
109:     {
110:         self::$widgetClasses[$name] = $class;
111:     }
112:     
113:     /**
114:      * Defines the class for a specific set of widgets, by name..
115:      *
116:      * @param string $classes
117:      *  The absolute name of the classes to define the widgets with,
118:      *  indexed by name.
119:      */
120:     public static function setWidgetClasses(array $classes, $merge = true)
121:     {
122:         self::$widgetClasses = $merge ?
123:             array_replace(self::$widgetClasses, $classes) : $classes;
124:     }
125: 
126:     /**
127:      * Constructor.
128:      *
129:      * @param array $configuration
130:      *  The express configuration array.
131:      */
132:     public function __construct(array $configuration = null)
133:     {
134:         parent::__construct($configuration);
135:     }
136: 
137:     /**
138:      * Defines the widget unique identifier.
139:      *
140:      * @param string $id
141:      *  The widget unique identifier.
142:      */
143:     public function setId($id)
144:     {       
145:         $this->id = $id;
146:     }
147: 
148:     /**
149:      * Returns the unique widget identifier, optionally generating one
150:      * based on an auto incrementable field.
151:      *
152:      * @param bool $generate
153:      *  When set to TRUE the ID will be generated before it is returned for use
154:      *  based on an incrementable field.
155:      *
156:      * @return string
157:      *  Returns the unique widget identifier.
158:      */
159:     public function getId($generate = true)
160:     {
161:         if ($generate && !isset($this->id))
162:         {
163:             $this->id = 'lw' . self::$nextWidgetId++;
164:         }
165: 
166:         return $this->id;
167:     }
168:     
169:     /**
170:      * Renders a context-less application view.
171:      *
172:      * @param string $view
173:      *  An absolute alias resolving to the view being rendered.
174:      *
175:      * @param array $variables
176:      *  The variables to be extracted into the script context.
177:      *
178:      * @param bool $capture
179:      *  When set to TRUE the rendered contents will be captured instead
180:      *  of sent to the currently active output buffer.
181:      *
182:      * @return string
183:      *  The captured contents, when applicable.
184:      */
185:     protected function render($view, array $variables = null, $capture = false)
186:     {
187:         $file = Lumina::getAliasPath($view, 'php', null);
188:         return View::getApplicationFileView($file, $variables)->run($capture);
189:     }
190:     
191:     /**
192:      * Builds the widget HTML element and returns it.
193:      *
194:      * @return HtmlElement
195:      *  The packed HTML element instance.
196:      */
197:     protected abstract function build();
198:     
199:     /**
200:      * Builds the widget HTML element and returns it.
201:      *
202:      * @return HtmlElement
203:      *  The packed HTML element instance.
204:      */
205:     public final function pack()
206:     {
207:         $element = $this->build();
208:         $element->setAttribute('id', $this->getId(true));
209:         return $element;
210:     }
211:     
212:     /**
213:      * Packs and renders this instance, optionally capturing any
214:      * generated contents.
215:      *
216:      * @param bool $capture
217:      *  When set to TRUE the rendered contents will be returned
218:      *  instead of sent to the currently active output buffer.
219:      *
220:      * @return string
221:      *  The rendered contents, if applicable.
222:      */
223:     public final function deploy($capture = false)
224:     {
225:         return $this->pack()->render($capture);
226:     }
227: }
