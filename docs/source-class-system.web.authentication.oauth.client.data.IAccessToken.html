 27: /**
 28:  * Represents an access token entry, that may be stored in a database and should
 29:  * atleast be associated with a resource owner, a client, and have an expiration date.
 30:  * 
 31:  * From RFC 6749 - The OAuth 2.0 Authorization Framework:
 32:  * 
 33:  * ACCESS TOKEN
 34:  *  Access tokens are credentials used to access protected resources. An access token
 35:  *  is a string representing an authorization issued to the client.
 36:  *  Tokens represent specific scopes and durations of access, granted by the 
 37:  *  resource owner, and enforced by the resource server and authorization server.
 38:  * 
 39:  * @author Igor Azevedo <igor.azevedo@neticle.pt>
 40:  * @since 0.2.0
 41:  */
 42: interface IAccessToken
 43: {
 44:     
 45:     /**
 46:      * Token is in good standing, not revoked or refreshed.
 47:      */
 48:     const STATUS_OK = 0;
 49:     
 50:     /**
 51:      * Token has been revoked by the authorization server and should no longer
 52:      * be considered valid.
 53:      */
 54:     const STATUS_REVOKED = 1;
 55:     
 56:     /**
 57:      * The refresh token has been used to generate a new access token,
 58:      * therefore, no more refreshes shall be allowed using the same refresh token.
 59:      */
 60:     const STATUS_REFRESHED = 2;
 61:     
 62:     /**
 63:      * The refresh token is invalid.
 64:      * Used when the reason for invalidation is not specified.
 65:      */
 66:     const STATUS_INVALID = 4;
 67:     
 68:     /**
 69:      * Token was issued with an authorization code.
 70:      */
 71:     const CONTEXT_RESOURCE_OWNER_ACCESS_TOKEN = 0;

 75:      */
 76:     const CONTEXT_RESOURCE_OWNER_IMPLICIT_ACCESS_TOKEN = 1;
 77:     
 78:     /**
 79:      * Token was issued to a client. (no owner)
 80:      */
 81:     const CONTEXT_CLIENT_ACCESS_TOKEN = 2;

 83:     const TYPE_BEARER = 'bearer';
