355:     /**
356:      * Runs this validation rule against the given model.
357:      *
358:      * @param Model $model
359:      *  The model being validated.
360:      *
361:      * @param string[] $attributes
362:      *  The names of the attributes to validate.
363:      *
364:      *  Please note that this rule will only validate the attribute it applies
365:      *  to, ignoring any extra arguments given in this array.
366:      *
367:      * @return bool
368:      *  Returns TRUE on success, FALSE on failure.
369:      */
370:     public function validate(IValidatableDataContainer $model, array $attributes = null)
371:     {
372:         $context = $model->getContext();
373:         
374:         if (!$this->contexts || in_array($context, $this->contexts))
375:         {
376:             // Determine the attributes to validate
377:             $attributes = $attributes ?
378:                 array_intersect($this->attributes, $attributes) : $this->attributes;
379:         
380:             $values = $model->getAttributes();
381:             $success = true;
382:         
383:             foreach ($attributes as $attribute)
384:             {
385:                 // The attribute value
386:                 $value = isset($values[$attribute]) ?
387:                     $values[$attribute] : null;
388:                 
389:                 // Make sure the attribute
390:                 if (empty($value) && ($value === null || $value === ''))
391:                 {
392:                     if ($this->required)
393:                     {
394:                         $success = false;
395:                         $this->reportEx($model, $attribute, $this->emptyMessage);
396:                     }
397:                 }
398:             
399:                 // Validate the attribute individually
400:                 else if (!$this->validateAttributeValue($model, $attribute, $value))
401:                 {
402:                     $success = false;
403:                 }
404:             
405:             }
406:             
407:             return $success;
408:         }
409:         
410:         return true;
411:     }
