 25: namespace system\web\authentication\oauth\server\role;

 27: /**
 28:  * Represents a client application and must be implemented by any class that 
 29:  * represents a third-party entity that will be making requests to protected
 30:  * resources while verified by the OAuth2 Provider.
 31:  * 
 32:  * From RFC 6749 - The OAuth 2.0 Authorization Framework:
 33:  * 
 34:  * CLIENT
 35:  *  An application making protected resource requests on behalf of the
 36:  *  resource owner and with its authorization. The term "client" does not imply
 37:  *  any particular implementation characteristics (e.g., whether the application 
 38:  *  executes on a server, a desktop, or other devices).
 39:  * 
 40:  * @author Igor Azevedo <igor.azevedo@neticle.pt>
 41:  * @since 0.2.0
 42:  */
 43: interface IClient extends IEntity
 44: {
 45:     
 46:     /**
 47:      * Clients capable of maintaining the confidentiality of their credentials, 
 48:      * or capable of secure client authentication using other means.
 49:      */
 50:     const TYPE_CONFIDENTIAL = 0;
 51:     
 52:     /**
 53:      * Clients incapable of maintaining the confidentiality of their credentials
 54:      * (e.g. clients executing on a device used by the resource owner, such as an 
 55:      * installed native application or a web browser-based application), and incapable
 56:      * of secure client authentication via any other means.
 57:      */
 58:     const TYPE_PUBLIC = 1;
 59:     
 60:     /**
 61:      * A web application is a confidential client running on a web server. 
 62:      * The client credentials as well as any access tokens issued to the client
 63:      * are stored on the web server and are not exposed to or accessible by the
 64:      * resource owner.
 65:      */
 66:     const PROFILE_WEB_APPLICATION = 0;
 67:     
 68:     /**
 69:      * A user-agent-based application is a public client in which the client code
 70:      * is downloaded from a web server and executes within an user-agent (e.g., a
 71:      * web browser) on the device used by the resource owner.
 72:      * Protocol data and credentials are easily accessible (and often visible) to
 73:      * the resource owner.
 74:      */
 75:     const PROFILE_UA_BASED_APPLICATION = 1;
 76:     
 77:     /**
 78:      * A native application is a public client installed and executed on the device 
 79:      * used by the resource owner. Protocol data and credentials are accessible to
 80:      * the resource owner. It is assumed that any client authentication 
 81:      * credentials included in the application can be extracted.
 82:      */
 83:     const PROFILE_NATIVE_APPLICATION = 2;
 84:         
 85:     /**
 86:      * Gets the client's type.
 87:      *
 88:      * This type is used to determine the level of safety the client is able to
 89:      * provide and therefore should return one of the values pre-defined as 
 90:      * constants on the IClient interface (TYPE_CONFIDENTIAL or TYPE_PUBLIC).
 91:      * 
 92:      * @return int
 93:      */
 94:     public function getType ();
 95:     
 96:     /**
 97:      * Gets the client's profile type.
 98:      * 
 99:      * @return int
100:      */
101:     public function getProfile ();
102:     
103:     /**
104:      * Gets the client's redirection endpoint URI.
105:      * 
106:      * This is where the authorization server will redirect the end-user to, along 
107:      * with the authorization code.
108:      * 
109:      * @return string
110:      */
111:     public function getRedirectionEndpointURI ();
112:     
113:     /**
114:      * Get's the client's secret key.
115:      * 
116:      * @return string
117:      *  The secret key, if any.
118:      */
119:     public function getSecret ();
120:     
121: }
