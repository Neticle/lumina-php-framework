 27: use \system\base\Module;
 28: use \system\core\ContextView;
 29: use \system\core\LazyExtension;
 30: use \system\core\exception\RuntimeException;

 32: /**
 33:  * A Context is a special kind of extension that can be used with other
 34:  * extensions capable of having different behaviours depending on the
 35:  * information provided by it.
 36:  *
 37:  * An example of where a Context extension is useful is when creating URLs
 38:  * that are relative to a specific module or controller.
 39:  *
 40:  * Context's also have renderable assets linked to them. This includes, between
 41:  * others, the application layouts and views.
 42:  *
 43:  * @author Lumina Framework <lumina@incubator.neticle.com>
 44:  * @since 0.2.0
 45:  */
 46: abstract class Context extends LazyExtension
 47: {
 48:     /**
 49:      * Constructor.
 50:      *
 51:      * @param string $name
 52:      *  The contextual extension name.
 53:      *
 54:      * @param Extension $parent
 55:      *  The parent extension instance, if any.
 56:      */
 57:     protected function __construct($name, Context $parent = null)
 58:     {
 59:         parent::__construct($parent);
 60:         $this->name = $name;
 61:     }
 62:     
 63:     /**
 64:      * Returns the context extension name.
 65:      *
 66:      * @return string
 67:      *  The extension name.
 68:      */
 69:     public final function getName()
 70:     {
 71:         return $this->name;
 72:     }
 73:     
 74:     /**
 75:      * Returns the route resolving to this context, relative to the
 76:      * base extension -- which is also a Context.
 77:      *
 78:      * @return string
 79:      *  The context route.
 80:      */
 81:     public final function getRoute()
 82:     {
 83:         $route = null;
 84:     
 85:         if (!$this->isBaseExtension())
 86:         {
 87:             $context = $this->getParent();
 88:             $route = $this->name;
 89:             
 90:             while (!$context->isBaseExtension())
 91:             {
 92:                 $route = $context->name . '/' . $route;
 93:                 $context = $context->getParent();
 94:             }
 95:         }
 96:         
 97:         return $route;
 98:     }
 99:     
100:     /**
101:      * This method is invoked right before a view is rendered.
102:      *
103:      * This method encapsulates the "render" event, which can not be canceled.
104:      *
105:      * @param View $view
106:      *  The View instance.
107:      *
108:      * @return bool
109:      *  Returns TRUE.
110:      */
111:     protected function onRender($view)
112:     {
113:         $this->raiseArray('render', [ $view ]);
114:         return true;
115:     }
116:     
117:     /**
118:      * This method is invoked right before a layout is rendered.
119:      *
120:      * This method encapsulates the "display" event, which can not be canceled.
121:      *
122:      * @param View $view
123:      *  The View instance.
124:      *
125:      * @return bool
126:      *  Returns TRUE.
127:      */
128:     protected function onDisplay($layout)
129:     {
130:         $this->raiseArray('display', [ $layout ]);
131:         return true;
132:     }
133:     
134:     /**
135:      * Returns the context layouts path.
136:      *
137:      * @param bool $recursive
138:      *  When set to TRUE the layout path will be retrieved recursively.
139:      *
140:      * @return string
141:      *  The context layouts path.
142:      */
143:     public abstract function getLayoutsPath();
144:     
145:     /**
146:      * Returns the absolute path to the layout script.
147:      *
148:      * @return string
149:      *  The absolute path to the layout script.
150:      */
151:     public abstract function getLayoutPath();
152:     
153:     /**
154:      * Returns the context views path.
155:      *
156:      * @return string
157:      *  The context views path.
158:      */
159:     public abstract function getViewsPath();
160:     
161:     /**
162:      * Renders a child view within the layout.
163:      *
164:      * @param string $view
165:      *  The name of the partial view to render.
166:      *
167:      * @param array $variables
168:      *  An associative array holding the variables to be extracted
169:      *  in the view context.
170:      *
171:      * @param bool $capture
172:      *  When set to TRUE the contents will be captured and returned
173:      *  instead of sent to the output buffer.
174:      *
175:      * @return string
176:      *  The captured contents, if applicable.
177:      */
178:     public function display($view, array $variables = null, $capture = false)
179:     {
180:         $layout = $this->getLayoutPath();
181:         
182:         if (isset($layout))
183:         {
184:             $layout = ContextView::getContextFileView
185:             (
186:                 $this, 
187:                 $layout,
188:                 [ 'viewContents' => $this->render($view, $variables, true) ]
189:             );
190:             
191:             $this->onDisplay($layout);          
192:             return $layout->run($capture);
193:         }
194:         
195:         throw new RuntimeException('Application layout is not defined.');
196:     }
197:     
198:     /**
199:      * Renders a child view.
200:      *
201:      * @param string $view
202:      *  The name of the partial view to render.
203:      *
204:      * @param array $variables
205:      *  An associative array holding the variables to be extracted
206:      *  in the view context.
207:      *
208:      * @param bool $capture
209:      *  When set to TRUE the contents will be captured and returned
210:      *  instead of sent to the output buffer.
211:      *
212:      * @return string
213:      *  The captured contents, if applicable.
214:      */
215:     public function render($view, array $variables = null, $capture = false)
216:     {
217:         $view = ContextView::getContextView($this, $view, $variables);
218:         $this->onRender($view); 
219:         return $view->run($capture);
220:     }
221:     
222:     /**
223:      * Resolves and returns the context route.
224:      *
225:      * If the route starts with '/' it is considered an absolute route; if
226:      * the route contains '/' at any other position it is considered relative
227:      * to the parent module, unless the context is already a module; 
228:      * if the route does not contain '/' it is considered relative to the
229:      * current context.
230:      *
231:      * @param string $route
232:      *  The route to be resolved.
233:      *
234:      * @return string
235:      *  The resolved absolute route.
236:      */
237:     public final function getResolvedContextRoute($route)
238:     {
239:         if (empty($route) || $route === '/')
240:         {
241:             return null;
242:         }
243:         
244:         if ($route[0] === '/')
245:         {
246:             return (isset($route[1])) ?
247:                 substr($route, 1) : null;
248:         }
249:         
250:         if ($this->isBaseExtension())
251:         {
252:             return $route;
253:         }
254:         
255:         if (($this instanceof Module) || strpos($route, '/') === false)
256:         {
257:             return $this->getRoute() . '/' . $route;
258:         }
259:         
260:         return $this->getParent()->getRoute() . '/' . $route;
261:     }
262:     
263:     /**
264:      * Forwards the current dispatch procedure to the specified route.
265:      *
266:      * @param string $route
267:      *  The route to dispatch to, relative to the current module.
268:      *
269:      * @param array $parameters
270:      *  An associative array defining the values to be bound to the action
271:      *  method, indexed by parameter name.
272:      *
273:      * @return bool
274:      *  Returns TRUE on success, FALSE otherwise.
275:      */
276:     protected function forward($route, array $parameters = null, $terminate = true)
277:     {
278:         $route = $this->getResolvedContextRoute($route);        
279:         $result = Lumina::getApplication()->dispatch($route, $parameters);
280:         
281:         if ($terminate)
282:         {
283:             exit;
284:         }
285:         
286:         return $result;
287:     }
288: }
