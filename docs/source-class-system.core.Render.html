 37: class Render extends Extension
 38: {   
 39:     /**
 40:      * Renders a script file.
 41:      *
 42:      * Please note the 'self' variable is not defined by default, meaning that
 43:      * you will have to call this function again if you wish to render a
 44:      * child view from that context.
 45:      *
 46:      * Make sure you can not render the view file from a controller before
 47:      * using this function.
 48:      *
 49:      * @param string $file
 50:      *  The absolute path to the file being rendered.
 51:      *
 52:      * @param array $variables
 53:      *  The variables to be extracted into the script context.
 54:      *
 55:      * @param bool $capture
 56:      *  When set to TRUE the rendered contents will be captured instead
 57:      *  of sent to the currently active output buffer.
 58:      */
 59:     protected static function renderFileEx($__FILE__, array $__VARIABLES__ = null, $__CAPTURE__ = true)
 60:     {
 61:         if (isset($__VARIABLES__))
 62:         {
 63:             extract($__VARIABLES__);
 64:         }
 65:         
 66:         if ($__CAPTURE__)
 67:         {
 68:             ob_start();
 69:             require($__FILE__);
 70:             return ob_get_clean();
 71:         }
 72:         
 73:         require($__FILE__);
 74:     }
 75:     
 76:     /**
 77:      * Renders a script file from this context.
 78:      *
 79:      * Make sure you can not render the view file from a controller before
 80:      * using this function.
 81:      *
 82:      * @param string $__FILE__
 83:      *  The absolute path to the file being rendered.
 84:      *
 85:      * @param array $__DATA__
 86:      *  The variables to be extracted into the script context.
 87:      *
 88:      * @param bool $__CAPTURE__
 89:      *  When set to TRUE the rendered contents will be captured instead
 90:      *  of sent to the currently active output buffer.
 91:      */
