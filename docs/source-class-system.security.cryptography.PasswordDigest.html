 30: /**
 31:  * The message digest component will apply a random salt to a given password
 32:  * and hash it according to certain options, trough PHP's 'crypt' 
 33:  * implementation.
 34:  *
 35:  * @author Lumina Framework <lumina@incubator.neticle.com>
 36:  * @since 0.2.0
 37:  */
 38: class PasswordDigest extends Component
 39: {
 40:     /**
 41:      * The default algorithm to use when digesting passwords.
 42:      *
 43:      * @type string
 44:      */
 45:     private $algorithm = 'blowfish';
 46:     
 47:     /**
 48:      * The options to apply when the default algorithm is being used
 49:      * to digest passwords.
 50:      *
 51:      * @type string
 52:      */
 53:     private $options = 
 54:     [
 55:         'cost' => 10
 56:     ];
 57:     
 58:     /**
 59:      * A static preffix salt to be applied to the raw passwords in order to
 60:      * make the generation of rainbow tables exceptionally hard, even when
 61:      * the attacker gains access to the original hash with lower cost.
 62:      *
 63:      * @type string
 64:      */
 65:     private $salt;
 66:     
 67:     /**
 68:      * Defines the default algorithm to use when digesting passwords.
 69:      *
 70:      * @param string $algorithm
 71:      *  The default algorithm to use when digesting passwords.
 72:      */
 73:     public function setAlgorithm($algorithm)
 74:     {
 75:         $this->algorithm = $algorithm;
 76:     }
 77:     
 78:     /**
 79:      * Returns the default algorithm to use when digesting passwords.
 80:      *
 81:      * @return string
 82:      *  The default algorithm to use when digesting passwords.
 83:      */
 84:     public function getAlgorithm()
 85:     {
 86:         return $this->algorithm;
 87:     }
 88:     
 89:     /**
 90:      * Defines the options to apply when the default algorithm is being used
 91:      * to digest passwords.
 92:      *
 93:      * @param array $options
 94:      *  The default algorithm options.
 95:      */
 96:     public function setOptions(array $options)
 97:     {
 98:         $this->options = $options;
 99:     }
100:     
101:     /**
102:      * Returns the options to apply when the default algorithm is being used
103:      * to digest passwords.
104:      *
105:      * @return array
106:      *  The default algorithm options.
107:      */
108:     public function getOptions()
109:     {
110:         return $this->options;
111:     }
112:     
113:     /**
114:      * Defines a static preffix salt to be applied to the raw passwords in order
115:      * to make the generation of rainbow tables exceptionally hard, even when
116:      * the attacker gains access to the original hash with lower cost.
117:      *
118:      * Please note that in addition to any defined salt, a random salt is always
119:      * generated and used when digesting passwords. Unlike that random salt,
120:      * this static salt will not be contained in the final hash as it's applied
121:      * to the passwords directly.
122:      *
123:      * @param string $salt
124:      *  The static password salt.
125:      */
126:     public function setSalt($salt)
127:     {
128:         $this->salt = $salt;
129:     }
130:     
131:     /**
132:      * Returns a static preffix salt to be applied to the raw passwords in order
133:      * to make the generation of rainbow tables exceptionally hard, even when
134:      * the attacker gains access to the original hash with lower cost.
135:      *
136:      * @return string
137:      *  The static password salt.
138:      */
139:     public function getSalt()
140:     {
141:         return $this->salt;
142:     }
143: 
144:     /**
145:      * Creates a random string to be used when building the salt.
146:      *
147:      * @param int $length
148:      *  The length of the random string to build.
149:      *
150:      * @return string
151:      *  The random string.
152:      */
153:     private function createRandomString($length)
154:     {
155:         $characters = 'abcdefghijklmnopqrstuvxywzABCDEFGHIJKLMNOPQRSTUVXYWZ0123456789.';
156:         $result = '';
157:         
158:         for ($i = 0; $i < $length; ++$i)
159:         {
160:             $index = rand(0, 62);
161:             $result .= $characters[$index];
162:         }
163:         
164:         return $result;
165:     }
166: 
167:     /**
168:      * Creates a random salt based on the algorithm and given options.
169:      *
170:      * @throws RuntimeException
171:      *  Thrown when the specified algorithm doesn't exist or lacks support
172:      *  from Lumina, PHP or the environment itself.
173:      *
174:      * @param string $algorithm
175:      *  The name of the algorithm to build the salt for.
176:      *
177:      * @param array $options
178:      *  An associative array containing the salt options.
179:      *
180:      * @return string
181:      *  The generated salt.
182:      */
183:     private function createRandomSalt($algorithm = 'blowfish', array $options = null)
184:     {
185:         if ($algorithm === 'blowfish' && CRYPT_BLOWFISH)
186:         {           
187:             if (isset($options['cost']))
188:             {
189:                 $cost = $options['cost'];
190:                 
191:                 if ($cost < 10)
192:                 {
193:                     $cost = '0' . $cost;
194:                 }
195:             }
196:             else
197:             {
198:                 $cost = '10';
199:             }
200:         
201:             return  '$2y$' . $cost . '$'
202:                 . $this->createRandomString(22);
203:         }
204:         
205:         else if ($algorithm === 'md5' && CRYPT_MD5)
206:         {
207:             return '$1$' . $this->createRandomString(9);
208:         }
209:         
210:         else if ($algorithm === 'sha256' && CRYPT_SHA256)
211:         {           
212:             $rounds = 'rounds=' . 
213:                 (
214:                     isset($options['rounds']) ?
215:                     ((string) $options['rounds']) : '5000'
216:                 ) . '$';
217:             
218:             return '$5$' . $rounds . $this->createRandomString(16 - strlen($rounds));
219:         }
220:         
221:         else if ($algorithm === 'sha512' && CRYPT_SHA512)
222:         {           
223:             $rounds = 'rounds=' . 
224:                 (
225:                     isset($options['rounds']) ?
226:                     ((string) $options['rounds']) : '5000'
227:                 ) . '$';
228:             
229:             return '$6$' . $rounds . $this->createRandomString(16 - strlen($rounds));
230:         }
231:         
232:         throw new RuntimeException('Unsupported algorithm "' . $algorithm . '" specified.');
233:     }
234:     
235:     /**
236:      * Digests a given password into a hash.
237:      *
238:      * @throws RuntimeException
239:      *  Thrown when the specified algorithm doesn't exist or lacks support
240:      *  from Lumina, PHP or the environment itself.
241:      *
242:      * @param string $password
243:      *  The password to digest.
244:      *
245:      * @param string $algorithm
246:      *  The algorithm to use when digesting the password, or NULL to use the
247:      *  configuration applied to this component.
248:      *
249:      * @param array $options
250:      *  An associative array containing the salt options.
251:      *
252:      * @return string
253:      *  The hashed password.
254:      */
255:     public function digest($password, $algorithm = null, array $options = null)
256:     {
257:         if (!isset($algorithm))
258:         {
259:             $algorithm = $this->algorithm;
260:             $options = $this->options;
261:         }
262:     
263:         return crypt($this->salt . $password, $this->createRandomSalt($algorithm, $options));
264:     }
265:     
266:     /**
267:      * Compares a raw password to a given hash.
268:      *
269:      * @param string $password
270:      *  The password to compare.
271:      *
272:      * @param string $hash
273:      *  The password hash, as returned by 'digest' method, which includes
274:      *  all the necessary information to match these two values.
275:      *
276:      * @return bool
277:      *  Returns TRUE if the password and hash match, FALSE otherwise.
278:      */
279:     public function compare($password, $hash)
280:     {
281:         return crypt($this->salt . $password, $hash) == $hash;
282:     }
283:     
284:     /**
285:      * Matches a given hash against the component configuration.
286:      *
287:      * If the hash does not match the component configuration, it should be
288:      * rehashed for security reasons.
289:      *
290:      * @throws RuntimeException
291:      *  Thrown if the hash fails to be parsed.
292:      *
293:      * @param string $hash
294:      *  The hash to be matched against the component.
295:      *
296:      * @return bool
297:      *  Returns TRUE if the hash matches, FALSE otherwise.
298:      */
299:     public function match($hash)
300:     {
301:         $info = $this->parse($hash);
302:         $algorithm = $info['algorithm'];
303:         
304:         if ($algorithm === $this->algorithm)
305:         {       
306:             if ($algorithm === 'blowfish')
307:             {
308:                 $cost = isset($this->options['cost']) ?
309:                     $this->options['cost'] : 10;
310:             
311:                 return ($cost === $info['cost']);
312:             }
313:         
314:             if ($algorithm === 'sha256' || $algorithm === 'sha512')
315:             {
316:                 $rounds = isset($this->options['rounds']) ?
317:                     $this->options['rounds'] : 5000;
318:             
319:                 return ($rounds === $info['rounds']);
320:             }
321:         }
322:         
323:         return false;
324:     }
325:     
326:     /**
327:      * Parses the given hash into an associative array containing the
328:      * following indexes:
329:      *
330:      *  "algorithm": the name of the algorithm used to generate the hash;
331:      *
332:      *  "modifier": the modifier, which might be the work cost or number
333:      *      of rounds required to generate the final hash;
334:      *
335:      * @throws RuntimeException
336:      *  Thrown if the hash fails to be parsed.
337:      *
338:      * @return array
339:      *  The parses hash information, as an associative array.
340:      */
341:     public function parse($hash)
342:     {
343:         $index = strpos($hash, '$', 1);
344:         
345:         if ($index)
346:         {
347:             $start = substr($hash, 0, $index);
348:         
349:             if ($start === '$1')
350:             {
351:                 return
352:                 [
353:                     'algorithm' => 'md5',
354:                     'modifier' => null
355:                 ];
356:             }
357:         
358:             if ($start === '$5' || $start === '$6')
359:             {
360:                 $rounds = (substr($hash, 2, 8) === '$rounds=') ?
361:                     intval(substr($hash, 10, strpos($hash, '$', $index + 1) - 10)) : 0;
362:             
363:                 return 
364:                 [
365:                     'algorithm' => $start === '$5' ? 'sha256' : 'sha512',
366:                     'modifier' => $rounds,
367:                     'rounds' => $rounds
368:                 ];
369:             }
370:             
371:             if ($start === '$2y' || $start === '$2x' || $start === '$2a')
372:             {
373:                 $cost = intval(ltrim(substr($hash, 4, 2), '0'));
374:             
375:                 return
376:                 [
377:                     'algorithm' => 'blowfish',
378:                     'modifier' => $cost,
379:                     'cost' => $cost
380:                 ];
381:             }
382:         }
383:         
384:         throw new RuntimeException('Failed to parse hash information.');
385:     }
386: }
