138:      * Constructor.
139:      *
140:      * @param Extension $parent
141:      *  The extension this view belongs to.

143:      * @param array $variables

179:     /**
180:      * Defines the variables to be extracted in the view script context.
181:      *
182:      * @param array $variables
183:      *  An associative array defining the variables, indexed by name.
184:      *
185:      * @param bool $merge
186:      *  When set to TRUE the variables will be merged with the ones
187:      *  previously defined.
188:      */
189:     public function setVariables(array $variables, $merge = false)
190:     {
191:         $this->variables = $merge ?
192:             array_replace($this->variables, $variables) : $variables;
193:     }

195:     /**
196:      * Renders a child view.
197:      *
198:      * @param string $view
199:      *  The name of the view to render.
200:      *
201:      * @param array $variables
202:      *  An associative array holding the variables to be extracted
203:      *  in the view context.
204:      *
205:      * @param bool $capture
206:      *  When set to TRUE the contents will be captured and returned
207:      *  instead of sent to the output buffer.
208:      *
209:      * @return string
210:      *  The captured contents, if applicable.
211:      */
212:     public function render($view, array $variables = null, $capture = false)
213:     {
214:         $nested = new View(Lumina::getAliasPath($view, 'php', $this->basePath));
215:         
216:         if (isset($variables))
217:         {
218:             $nested->setVariables($variables);
219:         }
220:         
221:         return $nested->run($capture);
222:     }
223:     
224:     /**
225:      * Runs the view by extracting the variables and including the script
226:      * file it is linked to.
227:      *
228:      * @param bool $capture
229:      *  When set to TRUE the contents will be captured and returned
230:      *  instead of sent to the output buffer.
231:      *
232:      * @return string
233:      *  The captured contents, if applicable.
234:      */
235:     public function run($capture = false)
236:     {
237:         return $this->renderFile($this->filePath, $this->variables, $capture);
238:     }
