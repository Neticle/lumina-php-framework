 30: /**
 31:  * An abstract paginator that provides a consistent API across multiple
 32:  * implementations.
 33:  *
 34:  * Although the base implementation is abstract, final implementations of this
 35:  * class will provide additional methods to ease the application of this
 36:  * feature.
 37:  *
 38:  * @author Lumina Framework <lumina@incubator.neticle.com>
 39:  * @since 0.2.0
 40:  */
 41: abstract class Paginator extends Element
 42: {
 43:     /**
 44:      * The instance of the provider this instance is linked with.
 45:      *
 46:      * @type Provider
 47:      */
 48:     private $provider;
 49:     
 50:     /**
 51:      * The number of items to display per page.
 52:      *
 53:      * @type int
 54:      */
 55:     private $interval = 25;
 56:     
 57:     /**
 58:      * The currently active page number.
 59:      *
 60:      * @type int
 61:      */
 62:     private $activePage = 1;
 63: 
 64:     /**
 65:      * Constructor.
 66:      *
 67:      * @param Provider $provider
 68:      *  The data provider this paginator is to be linked with.
 69:      *
 70:      * @param array $configuration
 71:      *  The paginator configuration.
 72:      */
 73:     protected function __construct(Provider $provider, array $configuration = null)
 74:     {
 75:         parent::__construct(null);
 76:         $this->provider = $provider;
 77:         $this->configure($configuration);
 78:     }
 79:     
 80:     /**
 81:      * Defines the number of items per page.
 82:      *
 83:      * @param int $interval
 84:      *  The number of items per page.
 85:      */
 86:     public function setInterval($interval)
 87:     {
 88:         if ($interval < 1)
 89:         {
 90:             throw new RuntimeException('Invalid value "' . $interval . '" specified for interval.');
 91:         }
 92:     
 93:         $this->interval = $interval;
 94:         $this->provider->reset();
 95:     }

 99:      *
100:      * @return int
101:      *  The number of items per page.
102:      */
103:     public function getInterval()
104:     {
105:         return $this->interval;
106:     }
107:     
108:     /**
109:      * Returns the provider this instance is linked to.
110:      *
111:      * @return Provider
112:      *  The provider this instance is linked to.
113:      */
114:     public function getProvider()
115:     {
116:         return $this->provider;
117:     }
118:     
119:     /**
120:      * Changes the paginator provider and defines the currently
121:      * active page back to 1.
122:      *
123:      * @param Provider $provider
124:      *  The new provider instance.
125:      */
126:     public function setProvider(Provider $provider)
127:     {
128:         $provider->reset();
129:         $this->provider = $provider;
130:         $this->activePage = 1;
131:     }
132:     
133:     /**
134:      * Returns the total item count, as reported by the
135:      * data provider.
136:      *
137:      * @return int
138:      *  The total item count, across all pages.
139:      */
140:     public function getTotalItemCount()
141:     {
142:         return $this->provider->getTotalItemCount();
143:     }
144:     
145:     /**
146:      * Returns the total page count.
147:      *
148:      * @return int
149:      *  The total page count.
150:      */
151:     public function getPageCount()
152:     {
153:         $itemCount = $this->provider->getTotalItemCount();
154:         
155:         if (isset($this->interval) && $this->interval > 0 && $itemCount > 0)
156:         {
157:             return (int) ceil($itemCount / $this->interval);
158:         }
159:         
160:         return 1;
161:     }
162:     
163:     /**
164:      * Defines the currently active page number.
165:      *
166:      * @throws RuntimeException

169:      * @param int $activePage
170:      *  The active page number.
171:      */
172:     public function setActivePage($activePage)
173:     {
174:         if ($activePage < 1 || $activePage > $this->getPageCount())
175:         {
176:             throw new RuntimeException('Invalid page "' . $activePage . '" specified.');
177:         }
178:         
179:         $this->activePage = $activePage;
180:         $this->provider->reset();
181:     }

183:     /**
184:      * Returns the currently active page number.
185:      *
186:      * @return int
187:      *  The currently active page number.
188:      */
189:     public function getActivePage()
190:     {
191:         return $this->activePage;
192:     }
193:     
194:     /**
195:      * Returns the offset for the first item of the specified page.
196:      *
197:      * @throws RuntimeException
198:      *  Thrown when the specified page number is invalid.
199:      *
200:      * @param int $page
201:      *  The page to get the offset for.
202:      *
203:      * @return int
204:      *  The page offset.
205:      */
206:     public function getPageOffset($page)
207:     {
208:         if ($page < 1 || $page > $this->getPageCount())
209:         {
210:             throw new RuntimeException('Invalid page "' . $activePage . '" specified.');
211:         }
212:         
213:         return ($page - 1) * $this->interval;
214:     }
215:     
216:     /**
217:      * Returns the offset for the first item of the currently active page.
218:      *
219:      * @return int
220:      *  The page offset.
221:      */
222:     public function getActivePageOffset()
223:     {
224:         return ($this->activePage - 1) * $this->interval;
225:     }
