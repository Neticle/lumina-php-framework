 53:     /**
 54:      * Creates a new SELECT statement.
 55:      *
 56:      * @param string $table
 57:      *  The name of the table to select from.
 58:      *
 59:      * @param Criteria|array $criteria
 60:      *  An instance of Criteria or it's express configuration array.
 61:      *
 62:      * @return Statement
 63:      *  The created Statement instance, ready for use.
 64:      */
 65:     public function createSelectStatement($table, $criteria = null)
 66:     {
 67:         $table = $this->quote($table);
 68:     
 69:         if (isset($criteria))
 70:         {
 71:             if (is_array($criteria))
 72:             {
 73:                 $criteria = new Criteria($criteria);
 74:             }
 75:             
 76:             $criteria = $criteria->toArray();
 77:             $statement = 'SELECT ';
 78:             
 79:             if ($criteria['distinct'])
 80:             {
 81:                 $statement .= 'DISTINCT ';
 82:             }
 83:             
 84:             // SELECT {fields}
 85:             $statement .= $criteria['select'] ?
 86:                 $criteria['select'] : '*';
 87:                 
 88:             // ... FROM {table} {alias}
 89:             $statement .= ' FROM ' . $table;
 90:             
 91:             if ($criteria['alias'])
 92:             {
 93:                 $statement .= ' ' . $this->quote($criteria['alias']);
 94:             }
 95:             
 96:             // ... JOIN
 97:             if ($criteria['join'])
 98:             {
 99:                 $statement .= ' ' . $criteria['join'];
100:             }
101:             
102:             // ... WHERE {condition}
103:             if ($criteria['condition'])
104:             {
105:                 $statement .= ' WHERE ' . $criteria['condition'];
106:             }
107:             
108:             // ... GROUP BY
109:             if ($criteria['group'])
110:             {
111:                 $statement .= ' GROUP BY ' . $criteria['group'];
112:             }
113:             
114:             // ... ORDER BY
115:             if ($criteria['sort'])
116:             {
117:                 $statement .= ' ORDER BY ' . $criteria['sort'];
118:             }
119:             
120:             // ... LIMIT
121:             if ($criteria['limit'])
122:             {
123:                 $statement .= ' LIMIT ' . $criteria['limit'];
124:             }
125:             
126:             // ... OFFSET
127:             if ($criteria['offset'])
128:             {
129:                 $statement .= ' OFFSET ' . $criteria['offset'];
130:             }
131:             
132:             // ... FOR
133:             if ($criteria['for'])
134:             {
135:                 if (!$this->getDriverConnection()->inTransaction())
136:                 {
137:                     throw new RuntimeException('Criteria property "for" can not be used without an ongoing transaction.');
138:                 }
139:             
140:                 if ($criteria['for'] === 'update')
141:                 {
142:                     $statement .= ' FOR UPDATE';
143:                 }
144:                 else if ($criteria['for'] === 'read')
145:                 {
146:                     $statement .= ' LOCK IN SHARE MODE';
147:                 }
148:             }
149:             
150:             $parameters = $criteria['parameters'];
151:             
152:         }
153:         else
154:         {
155:             $statement = 'SELECT * FROM ' . $table;
156:             $parameters = null;
157:         }
158:         
159:         return $this->prepare($statement, $parameters);
160:     }

223:     /**
224:      * Creates a new UPDATE statement.
225:      *
226:      * @param string $table
227:      *  The name of the table to select from.
228:      *
229:      * @param array $fields
230:      *  An associative array holding the values to define, indexed by
231:      *  field name.
232:      *
233:      * @param Criteria|array $criteria
234:      *  An instance of Criteria or it's express configuration array.
235:      *
236:      * @return Statement
237:      *  The created Statement instance, ready for use.
238:      */
239:     public function createUpdateStatement($table, array $fields, $criteria = null)
240:     {
241:         $definitions = [];
242:         $parameters = [];
243:         
244:         foreach ($fields as $field => $value)
245:         {   
246:             if ($value instanceof Expression)
247:             {
248:                 $value = $value->toString();
249:             }
250:             else
251:             {
252:                 $key = Criteria::getUniqueParameterIdentifier();
253:                 $parameters[$key] = $value;
254:                 $value = $key;
255:             }
256:             
257:             $definitions[] = $this->quote($field) . '=' . $value;
258:         }
259:         
260:         $table = $this->quote($table);
261:         $definitions = implode(', ', $definitions);
262:         
263:         if (isset($criteria))
264:         {
265:             if (is_array($criteria))
266:             {
267:                 $criteria = new Criteria($criteria);
268:             }
269:             
270:             $criteria = $criteria->toArray();
271:             
272:             // UPDATE {table} {alias}
273:             $statement = 'UPDATE ' . $table;
274:             
275:             if ($criteria['alias'])
276:             {
277:                 $statement .= ' ' . $criteria['alias'];
278:             }
279:             
280:             // ... SET 
281:             $statement .= ' SET ' . $definitions;
282:             
283:             // ... JOIN
284:             if ($criteria['join'])
285:             {
286:                 $statement .= ' ' . $criteria['join'];
287:             }
288:             
289:             // ... WHERE {condition}
290:             if ($criteria['condition'])
291:             {
292:                 $statement .= ' WHERE ' . $criteria['condition'];
293:             }
294:             
295:             if ($criteria['parameters'])
296:             {
297:                 $parameters += $criteria['parameters'];
298:             }
299:         }
300:         else
301:         {
302:             $statement = 'UPDATE ' . $table . ' SET ' . $definitions;
303:         }
304:         
305:         return $this->prepare($statement, $parameters);
306:     }

326:

337:                 $value = $key;

339:

344:         $statement = 'INSERT INTO ' . $this->quote($table)

351:     /**
352:      * Creates a new statement to call a stored procedure.
353:      *
354:      * Although Lumina does translate all named parameters to numeric
355:      * parameters, you can still use named parameters to increase your code
356:      * readability without any performance impact.
357:      *
358:      * @param string $procedure
359:      *  The name of the procedure to call.
360:      *
361:      * @param array $parameters
362:      *  The parameters to be bound to the call statement, in the same order
363:      *  as they are declared by the procedure.
364:      */
365:     public function createCallProcedureStatement($procedure, array $parameters = null)
366:     {
367:         $statement = 'CALL ' . $this->quote($procedure) . '(';
368:         
369:         if (isset($parameters))
370:         {
371:             $collection = [];
372:             $values = [];
373:             $i = 0;
374:         
375:             foreach ($parameters as $name => $value)
376:             {
377:                 if ($value instanceof Expression)
378:                 {
379:                     $values[] = $value->toString();
380:                     continue;
381:                 }
382:                 
383:                 $values[] = '?';
384:                 $collection[++$i] = $value;
385:             }
386:             
387:             $statement .= implode(', ', $values);
388:             $parameters = $collection;
389:         }
390:         
391:         $statement .= ')';
392:         
393:         return $this->prepare($statement, $parameters);
394:     }
