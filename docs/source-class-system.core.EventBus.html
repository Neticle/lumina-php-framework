 25: namespace system\core;
 26: 
 27: /**
 28:  * A centralized event bus capable of having handlers registered to events
 29:  * raised by multiple class instances that might not even exist and the moment.
 30:  *
 31:  * @author Lumina Framework <lumina@incubator.neticle.com>
 32:  * @since 0.2.0
 33:  */
 34: class EventBus
 35: {
 36:     /**
 37:      * An multi-indexed array holding the handler callable reference, indexed
 38:      * by class and event.
 39:      *
 40:      * @type array
 41:      */
 42:     private $handlers = [];
 43: 
 44:     /**
 45:      * Constructor.
 46:      */
 47:     public function __construct()
 48:     {
 49:     }
 50:     
 51:     /**
 52:      * Registers a new event handler for the specified event, regardless of
 53:      * which object it was raised from.
 54:      *
 55:      * @param string $event
 56:      *  The name of the event to register the handler for.
 57:      *
 58:      * @param callable $callback
 59:      *  A callable reference to the function handling this event.
 60:      *
 61:      *  The function signature depends on the event it is supposed to handle and
 62:      *  you should look at the element documentation to find out exactly what
 63:      *  it is.
 64:      *
 65:      *  The callback function should always return TRUE, unless it needs to
 66:      *  stop propagation or cancel the event, in which case FALSE should be
 67:      *  returned.
 68:      *
 69:      *  Do not forget NULL evaluates to FALSE, meaning that if no return
 70:      *  statement is specified no more handlers will be called and in some
 71:      *  cases the event will be canceled.
 72:      */
 73:     public function on($event, callable $callback)
 74:     {
 75:         return $this->onClassEvent(0, $event, $callback);
 76:     }
 77:     
 78:     /**
 79:      * Registers a new event handler for the specified event, as long as it
 80:      * is raised by an object of the specified class.
 81:      *
 82:      * @param string $class
 83:      *  The name of the class to register the event handler for.
 84:      *
 85:      * @param string $event
 86:      *  The name of the event to register the handler for.
 87:      *
 88:      * @param callable $callback
 89:      *  A callable reference to the function handling this event.
 90:      *
 91:      *  The function signature depends on the event it is supposed to handle and
 92:      *  you should look at the element documentation to find out exactly what
 93:      *  it is.
 94:      *
 95:      *  The callback function should always return TRUE, unless it needs to
 96:      *  stop propagation or cancel the event, in which case FALSE should be
 97:      *  returned.
 98:      *
 99:      *  Do not forget NULL evaluates to FALSE, meaning that if no return
100:      *  statement is specified no more handlers will be called and in some
101:      *  cases the event will be canceled.
102:      */
103:     public function onClassEvent($class, $event, callable $callback)
104:     {
105:         if (!isset($this->handlers[$event]))
106:         {
107:             $this->handlers[$event] = [];
108:         }
109:         
110:         $this->handlers[$event][$class][] = $callback;
111:     }
112:     
113:     /**
114:      * Raises the specified event through this event bus.
115:      *
116:      * @param object $source
117:      *  The object this event is being raised from.
118:      *
119:      * @param string $event
120:      *  The event to raise.
121:      *
122:      * @param mixed ...$arguments
123:      *  The event arguments.
124:      *
125:      * @return bool
126:      *  Returns FALSE if one of the handlers stopped propagation or canceled
127:      *  the event, TRUE otherwise.
128:      */
129:     public function raise($source, $event)
130:     {
131:         return $this->raiseArray($source, $event, array_slice(func_get_args(), 2));
132:     }
133:     
134:     /**
135:      * Raises the specified event through this event bus.
136:      *
137:      * @param object $source
138:      *  The object this event is being raised from.
139:      *
140:      * @param string $event
141:      *  The event to raise.
142:      *
143:      * @param array $arguments
144:      *  The event arguments.
145:      *
146:      * @return bool
147:      *  Returns FALSE if one of the handlers stopped propagation or canceled
148:      *  the event, TRUE otherwise.
149:      */
150:     public function raiseArray($source, $event, array $arguments = null)
151:     {
152:         if (isset($this->handlers[$event]))
153:         {
154:             $callbackArguments = null;
155:             
156:             foreach ($this->handlers[$event] as $class => $handlers)
157:             {
158:                 if ($class === 0 || is_a($source, $class))
159:                 {
160:                     if(!isset($callbackArguments))
161:                     {
162:                         $callbackArguments = array_merge([ $source ], (array) $arguments);
163:                     }
164:                     
165:                     foreach ($handlers as $callback)
166:                     {
167:                         if (!call_user_func_array($callback, $callbackArguments))
168:                         {
169:                             return false;
170:                         }
171:                     }
172:                 }
173:             }
174:         }
175:         
176:         return true;
177:     }
178: }
