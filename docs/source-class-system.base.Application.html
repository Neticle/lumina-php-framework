 47:     /**
 48:      * This method is invoked during the application construction procedure,
 49:      * before the configuration takes place.
 50:      *
 51:      * This method encapsulates the "construction" event.
 52:      *
 53:      * @return bool
 54:      *  Returns TRUE to continue with the event, FALSE to cancel it.
 55:      */
 56:     protected function onConstruction()
 57:     {
 58:         if (parent::onConstruction())
 59:         {       
 60:             // Register the core components
 61:             $this->setComponents
 62:             (
 63:                 
 64:                 [
 65:                     // Handles all interactions between the application elements
 66:                     // and a SQL database.
 67:                     'database' => 
 68:                     [
 69:                         'class' => 'system\\sql\\Connection',
 70:                         'driver' => 'mysql'
 71:                     ],
 72:                 
 73:                     // Handles translation of messages and number formating for
 74:                     // multi-locale application.
 75:                     'dictionary' => 
 76:                     [
 77:                         'class' => 'system\\i18n\\dictionary\\StaticDictionary',
 78:                         'locale' => 'en_GB'
 79:                     ],
 80:                     
 81:                     // Manages cached entries in order to avoid running further
 82:                     // expensive deterministic procedures in future requests.
 83:                     'cache' => 
 84:                     [
 85:                         'class' => 'system\\cache\\DefaultCache'
 86:                     ],
 87:                     
 88:                     // Implements an HTTP client capable of sending requests and 
 89:                     // parsing the responses.
 90:                     'http' => 
 91:                     [
 92:                         'class' => 'system\\http\\HttpComponent',
 93:                         'wrapper' => 'curl'
 94:                     ]
 95:                 
 96:                 ]
 97:             );
 98:             
 99:             return true;
100:         }
101:         
102:         return false;
103:     }
104:     
105:     /**
106:      * Defines or updates the application context.
107:      *
108:      * @param Context $context
109:      *  The application context, or NULL to remove.
110:      */
